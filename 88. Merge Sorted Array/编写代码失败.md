public class Solution {
    public void Merge(int[] nums1, int m, int[] nums2, int n) {
        int i=0,j=0;
        if(n==0)
        {
            return;
        }
        else if(m==0)
        {
            while(i<m+n&&j<n)
            {
                nums1[i++]=nums2[j++];
            }
        }
        else
        {
            while(i<m+n&&j<n)
            {
                if(nums1[i]==0&&m==0)
                {
                    while(j<n)
                    {
                        nums1[i++]=nums2[j++];
                    }
                    break;
                }            
                else if(nums1[i]>nums2[j])
                {
                    for(int a=m+j;a>=i;a--)
                    {
                        if(a==i)
                        {
                            nums1[a]=nums2[j];
                        }
                        else
                        {
                            nums1[a]=nums1[a-1];
                        }
                    }
                    i++;
                    j++;
                }
                else if(nums1[i]<nums2[j])      //重点
                {
                    if(nums1[i]==0&&(i>=m)||(m<=n)||((m+n-i)==(n-j))))
                    {
                        nums1[i]=nums2[j];
                        i++;
                        j++;
                    }
                    else 
                    {
                        i++;
                    }
                }
                else if(nums1[i]==nums2[j])
                {
                    for(int a=m+j;a>i;a--)
                    {
                        if(a==i+1)
                        {
                            nums1[a]=nums2[j];
                        }
                        else
                        {
                            nums1[a]=nums1[a-1];
                        }
                    }
                    i=i+2;
                    j++;
                }
            }
        }

    }
}
